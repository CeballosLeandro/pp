CREATE TABLE `museo`.`objeto` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(100) NOT NULL , `especialidad` VARCHAR(10) NOT NULL , `ubicacion` INT(50) NOT NULL , `sector` VARCHAR(50) NOT NULL , `descripcion` TEXT NOT NULL , `descripcionh` TEXT NOT NULL , `fechaAlta` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB;
CREATE TABLE `museo`.`usuario` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(100) NOT NULL , `mail` VARCHAR(100) NOT NULL , `password` VARCHAR(255) NOT NULL , PRIMARY KEY (`id`), UNIQUE (`mail`)) ENGINE = InnoDB;
CREATE TABLE `museo`.`historial` (`id` INT(11) NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(45) NOT NULL , `especialidad` VARCHAR(12) NOT NULL , `ubicacion` INT(255) NOT NULL , `sector` VARCHAR(150) NOT NULL , `descripcion` TEXT NOT NULL , `descripcionH` TEXT NOT NULL , `fechaAlta` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB;
